矩陣法求線性回歸¶

#設定資訊

# 因子'LSTAT' 的訓練資料
AA=list(X_train['LSTAT']) 

# 因子'RM' 的訓練資料
BB=list(X_train['RM'])

# 結果 Y_train 的訓練資料
CC=list(Y_train)

# 因子'LSTAT' 的測試資料
DD=list(X_test['LSTAT'])

# 因子'RM' 的測試資料
EE=list(X_test['RM'])

# 結果 Y_train 的測試資料
FF=list(Y_test)

import numpy as np
import matplotlib.pyplot as plt


X = np.matrix([np.ones(404), AA, BB]).T   # 因為影響房價有兩個因子所以 AA 和 BB 要一同算進去
y = np.matrix(CC).T

# Solve for projection matrix
p_mat = np.linalg.inv(X.T.dot(X)).dot(X.T).dot(y)  # 得到線性方程式的係數
print(p_mat)

[[ 2.73624034]
 [-0.71722954]
 [ 4.58938833]]

# 打分數


y_train_predict = np.array(p_mat[0] + p_mat[1] * AA + p_mat[2]*BB ).tolist()[0] # 帶回係數算出Y訓練的預測值 
rmse_M_L = (np.sqrt(mean_squared_error(CC, y_train_predict)))  #評分
r2_M_L = sklearn.metrics.r2_score(CC, y_train_predict)

print("The model performance for training set (matrix)")
print("--------------------------------------")
print('RMSE is {}'.format(rmse_M_L))
print('R2 score is {}'.format(r2_M_L))
print("\n")

y_test_predict = np.array(p_mat[0] + p_mat[1] * DD + p_mat[2]*EE ).tolist()[0] # 帶回係數算出Y測試的預測值 
rmse_M_T= (np.sqrt(mean_squared_error(FF, y_test_predict)))  #評分
r2_M_T= sklearn.metrics.r2_score(FF, y_test_predict)

print("The model performance for testing set (matrix)")
print("--------------------------------------")
print('RMSE is {}'.format(rmse_M_T))
print('R2 score is {}'.format(r2_M_T))
